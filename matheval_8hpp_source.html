97 /** @brief Abstract Syntax Tree
98  *
99  * Stores the abstract syntax tree (AST) of the parsed mathematical
100  * expression.
101  */
102 template < typename real_t >
103 struct expr_ast
104 {
105  using tree_t = boost::variant<
106  nil // can't happen!
107  , real_t
108  , std::string
109  , boost::recursive_wrapper<expr_ast<real_t>>
110  , boost::recursive_wrapper<binary_op<real_t>>
111  , boost::recursive_wrapper<unary_op<real_t>>
112  >;
113 public:
114  /** @brief AST storage
115  *
116  * The syntax tree can hold various types. Numbers (`real_t`),
117  * variables (`std::string`), the recursive tree itself
118  * (`expr_ast`), binary operators (`binary_op`), and unary
119  * operators (`unary_op`).
120  */
121  tree_t tree;
122 
123  /** @brief Default constructor
124  *
125  * Initializes the tree to a nil value to indicate inconsistent
126  * state.
127  */
128  expr_ast() : tree(nil{}) {}
129 
130  /** @brief Copy constructor
131  *
132  * Deep copies the syntax tree.
133  */
134  template <typename Expr>
135  expr_ast(Expr other) : tree(std::move(other)) {} // NOLINT
136 
137  /** @brief Add a tree */
138  expr_ast& operator+=(expr_ast const &rhs);
139  /** @brief subtract a tree */
140  expr_ast& operator-=(expr_ast const &rhs);
141  /** @brief Multiply by a tree */
142  expr_ast& operator*=(expr_ast const &rhs);
143  /** @brief Divide by a tree */
144  expr_ast& operator/=(expr_ast const &rhs);

210 /** @brief Evaluate the Abstract Syntax Tree
211  *
212  * This visits all the variants of the AST and applies the stored
213  * operators.
214  */
215 template < typename real_t >
216 class eval_ast
217 {
218 public:
219  /** @brief Necessary typedef for `boost::apply_visitor` */
220  using result_type = real_t;
221 
222  /** @brief Type of the symbol table */
223  using symbol_table_t = std::map<std::string, result_type>;
224 
225  /** @brief Constructor
226  *
227  * Saves the symbol table to apply variables.
228  */
229  explicit eval_ast(symbol_table_t sym) : st(std::move(sym)) {}
230 
231  /** @brief Empty nodes in the tree evaluate to 0 */
232  result_type operator()(nil /*unused*/) const { return 0; }
233 
234  /** @brief Numbers evaluate to themselves */
235  result_type operator()(result_type n) const { return n; }
236 
237  /** @brief Variables evaluate to their value in the symbol table */
238  result_type operator()(std::string const &c) const
239  {
240  auto it = st.find(c);
241  if (it == st.end()) {
242  throw std::invalid_argument("Unknown variable " + c); // NOLINT
243  }
244  return it->second;
245  }
246 
247  /** @brief Recursively evaluate the AST */
248  result_type operator()(expr_ast<real_t> const& ast) const
249  {
250  return boost::apply_visitor(*this, ast.tree);
251  }
252 
253  /** @brief Evaluate a binary operator and optionally recurse its operands */
254  result_type operator()(binary_op<real_t> const& tree) const
255  {
256  return tree.op(
257  boost::apply_visitor(*this, tree.lhs.tree),
258  boost::apply_visitor(*this, tree.rhs.tree)
259  );
260  }
261 
262  /** @brief Evaluate a unary operator and optionally recurse its operand */
263  result_type operator()(unary_op<real_t> const& tree) const

265  return tree.op(
266  boost::apply_visitor(*this, tree.rhs.tree)

268  }

270 private:
271  symbol_table_t st;
272 };

287 template <typename real_t> struct ConstantFolder {
288  /** @brief Necessary typedef for `boost::apply_visitor` */
289  using result_type = typename expr_ast<real_t>::tree_t;
290 
291  /** @brief Empty nodes in the tree evaluate to 0 */
292  result_type operator()(nil /*unused*/) const { return 0; }
293 
294  /** @brief Numbers evaluate to themselves */
295  result_type operator()(real_t n) const { return n; }
296 
297  /** @brief Variables do not evaluate */
298  result_type operator()(std::string const &c) const { return c; }
299 
300  /** @brief Recursively evaluate the AST */
301  result_type operator()(expr_ast<real_t> const &ast) const {
302  return boost::apply_visitor(*this, ast.tree);
303  }
304 
305  /** @brief Evaluate a binary operator and optionally recurse its operands */
306  result_type operator()(binary_op<real_t> const &tree) const {
307  auto lhs = boost::apply_visitor(*this, tree.lhs.tree);
308  auto rhs = boost::apply_visitor(*this, tree.rhs.tree);
309 
310  /* If both operands are known, we can directly evaluate the function,
311  * else we just update the children with the new expressions. */
312  if (holds_alternative<real_t>(lhs) && holds_alternative<real_t>(rhs)) {
313  return tree.op(boost::get<real_t>(lhs), boost::get<real_t>(rhs));
314  }
315  return binary_op<real_t>(tree.op, lhs, rhs);
316  }
317 
318  /** @brief Evaluate a unary operator and optionally recurse its operand */
319  result_type operator()(unary_op<real_t> const &tree) const {
320  auto rhs = boost::apply_visitor(*this, tree.rhs.tree);
321  /* If the operand is known, we can directly evaluate the function. */
322  if (holds_alternative<real_t>(rhs)) {
323  return tree.op(boost::get<real_t>(rhs));
324  }
325  return unary_op<real_t>(tree.op, rhs);
326  }

514  >> *( ("**" > factor [_val = binary_expr(pow, _val, _1)])

516  ;

520

553 /** @brief Parse an expression
554  *
555  * This function builds the grammar and parses the iterator into
556  * an AST.
557  *
558  * @param[in] first iterator to the start of the input sequence
559  * @param[in] last iterator to the end of the input sequence
560  */
561 template <typename real_t, typename Iterator>
562 detail::expr_ast<real_t> parse(Iterator first, Iterator last) {
563  static detail::grammar<real_t, Iterator> const g;
564 
565  auto ast = detail::expr_ast<real_t>{};
566 
567  bool r = qi::phrase_parse(first, last, g, ascii::space, ast);
568 
569  if (!r || first != last) {
570  std::string rest(first, last);
571  throw std::runtime_error("Parsing failed at " + rest); // NOLINT
572  }
573 
574  return ast;
575 }
